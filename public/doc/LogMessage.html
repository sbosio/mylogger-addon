def initialize(logplex_message)
  fields = logplex_message.match(MESSAGE_FORMAT).to_a

  @priority = fields[1].to_i
  @timestamp = Time.parse(fields[2])
  @host = fields[3]
  @application = fields[4]
  @process = fields[5]
  @message = fields[6]
end
